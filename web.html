HTML Best Practices

  • Always declare DOCTYPE
  • Use semantic HTML elements
  • Include alt attributes for images
  • Use proper heading hierarchy (h1-h6)
  • Validate your HTML
  • Ensure accessibility with ARIA when needed
  • Keep nesting minimal and clean
  • Use UTF-8 character encoding
  • Include viewport meta tag for mobile
  • Use lowercase for element names

3. Operators

3.1 Arithmetic Operators

let x = 10, y = 3;

x + y;  // 13 (Addition)
x - y;  // 7 (Subtraction)
x * y;  // 30 (Multiplication)
x / y;  // 3.333... (Division)
x % y;  // 1 (Remainder)
x ** y; // 1000 (Exponentiation)
x++;    // 11 (Post-increment)
++x;    // 12 (Pre-increment)

3.2 Comparison Operators

10 == '10';   // true (loose equality)
10 === '10';  // false (strict equality)
10 != '10';   // false
10 !== '10';  // true
5 > 3;        // true
5 <= 5;       // true

3.3 Logical Operators

true && false;  // false (AND)
true || false;  // true (OR)
!true;          // false (NOT)

// Short-circuit evaluation
false && console.log('Never runs');
true || console.log('Never runs');

3.4 Other Operators

// Ternary operator
let status = (age >= 18) ? 'Adult' : 'Minor';

// Nullish coalescing
let value = null ?? 'default'; // 'default'

// Optional chaining
user?.address?.street;

5. Functions

5.1 Function Declarations

// Function declaration
function greet(name) {
    return `Hello, ${name}!`;
}

// Function expression
const greet = function(name) {
    return `Hello, ${name}!`;
};

// Arrow function
const greet = (name) => `Hello, ${name}!`;

5.2 Parameters & Arguments

// Default parameters
function greet(name = 'Guest') {
    return `Hello, ${name}!`;
}

// Rest parameters
function sum(...numbers) {
    return numbers.reduce((a, b) => a + b);
}

// Destructuring parameters
function printUser({name, age}) {
    console.log(`${name} is ${age} years old`);
}

5.3 Higher-Order Functions

// Returning functions
function multiplier(factor) {
    return function(number) {
        return number * factor;
    };
}

    return operation(a, b);

17. JavaScript Best Practices

17.1 Code Organization

  • Use modules to organize code
  • Follow consistent naming conventions (camelCase for variables/functions, PascalCase for classes)
  • Keep functions small and focused
  • Use comments judiciously

17.2 Performance

  • Minimize DOM access and manipulation
  • Debounce or throttle event handlers for frequent events
  • Use event delegation for dynamic elements
  • Avoid memory leaks by cleaning up event listeners

17.3 Security

  • Always validate and sanitize user input
  • Use HTTPS for API requests
  • Be cautious with eval() and innerHTML
  • Implement proper error handling
